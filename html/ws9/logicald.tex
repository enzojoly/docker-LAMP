@startuml
!theme plain
!includeurl https://raw.githubusercontent.com/plantuml-stdlib/C4-PlantUML/master/C4_Container.puml

entity "Cities" as cities {
  * City ID (PK) : int
  --
  * Name : varchar
  * Country : varchar
  * Population : int
  * Geo-location (Long x Lat) : varchar
  * Currency : varchar
  * Mayor : varchar
  * Year Founded : int
}

entity "Place of Interest" as places {
  * Place ID (PK) : int
  --
  * Name : varchar
  * Type : varchar
  * Capacity : int
  * Geo-location : varchar
  * Photos (flickr API) : varchar
  * Opening hours : varchar
  * Description : text
  * City ID (FK referencing City) : int
}

entity "Currency" as currency {
  * Currency ID (PK) : int
  --
  * Currency Name : varchar
  * Currency Code : varchar
  * Exchange Rate : decimal
  * Symbol : varchar
  * Country ID (FK referencing City) : int
}

entity "Weather" as weather {
  * Weather ID (PK) : int
  --
  * Temperature : decimal
  * Humidity : decimal
  * Wind Speed : decimal
  * Wind Direction : varchar
  * Weather Condition : varchar
  * City ID (FK referencing City) : int
}

entity "Category" as category {
  * Category ID (PK) : int
  --
  * Category Name : varchar
}

entity "Place_Category" as place_category {
  * Place Category ID (PK) : int
  --
  * Category ID : int (FK referencing Category)
  * Place ID (FK referencing Place) : int
}
weather ||--|| cities : "affects"
places }o--|| cities : "located in"
currency ||--|| cities : "used in"
places ||--|{ place_category : "categorised as"
place_category }|--|| category : "is"
@enduml
